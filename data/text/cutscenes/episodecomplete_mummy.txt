// HELPER FUNCTIONS

function signanimate(sign) {
    var s = sign.show("turn");
    //s.onComplete(function() { resume(); sign.show("loop"); });
    //pause();
    while(!s.completed) {
        // do nothing
    }
    sign.show("loop");
}

function spinthewheel(sign, wheel, animation) {
  var s = sign.show("turn");
    //s.onComplete(function() { resume(); sign.show("loop"); });
    //pause();
    while(!s.completed) {
        // do nothing
    }
    sign.hide();

  wheel.x = 30;
  wheel.y = -30;
  play("spinthewheel");
    var s = wheel.show(animation);
    //s.onComplete(function() { resume(); sign.show("loop"); });
    //pause();
    while(!s.completed) {
        // do nothing
    }
    wheel.hide();
}
